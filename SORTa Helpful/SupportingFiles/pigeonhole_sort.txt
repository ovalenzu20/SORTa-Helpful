func sort(arr):
    minElement = min(arr)
    maxElement = max(arr)
    size = maxElement - minElement + 1

    // Array containing zeroes
    holes = [0] * size

    for x in arr:
        holes[x - minElement] += 1

    i = 0

    for count in range(0, size):
        while holes[count] > 0:
            holes[count] -= 1
            arr[i] = count + minElement
            i += 1
