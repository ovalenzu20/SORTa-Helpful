func sort(arr):
    n = len(arr)
    // Initialize gap
    gap = n
    swapped = True

    while gap != 1 or swapped == True:
        gap = getNextGap(gap)
        swapped = False
        for i in range(0, n - gap):
            if arr[i] > arr[i + gap]:
                swap(arr[i], arr[i + gap])
                swapped = True

func getNextGap(gap):
    gap = (gap * 10) / 13
    if gap < 1:
        return 1
    return gap
